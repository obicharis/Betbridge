function setDir(dir) { currentDir = dir; document.getElementById('btn-sb-st').classList.toggle('active', dir === 'sb-st'); document.getElementById('btn-st-sb').classList.toggle('active', dir === 'st-sb'); const input = document.getElementById('code-input'); const label = document.getElementById('input-label'); const hint = document.getElementById('hint-text'); if (dir === 'sb-st') { label.textContent = 'SportyBet Share Code'; input.placeholder = 'e.g. SBNG1234XYZ'; hint.textContent = 'Paste your SportyBet share code above'; } else { label.textContent = 'Stake Share Code or Link'; input.placeholder = 'e.g. ABC123XY or full stake.com link'; hint.textContent = 'Paste your Stake share code or full share link above'; } // Reset result document.getElementById('result-card').classList.remove('visible'); document.getElementById('error-box').classList.remove('visible'); input.value = ''; } async function convert() { const code = document.getElementById('code-input').value.trim(); if (!code) { showError('Please paste a share code first.'); return; } setLoading(true); hideResult(); hideError(); try { const endpoint = currentDir === 'sb-st' ? `/api/sportybet-to-stake?code=${encodeURIComponent(code)}` : `/api/stake-to-sportybet?code=${encodeURIComponent(code)}`; const res = await fetch(endpoint); const data = await res.json(); if (!res.ok || !data.success) { showError(data.error || 'Something went wrong. Please try again.'); return; } showResult(data); } catch (err) { showError('Network error. Please check your connection and try again.'); } finally { setLoading(false); } } function showResult(data) { const { betslip, resolution, targetPlatform } = data; const sel = resolution?.selections || betslip.selections; const deeplink = resolution?.deeplink || null; const matchRate = resolution?.matchRate ?? 1; const matchedCount = resolution ? sel.filter(s => s.status === 'matched').length : sel.length; // Title document.getElementById('result-title').textContent = `${sel.length} Selection${sel.length !== 1 ? 's' : ''} · ${betslip.betType.toUpperCase()}`; // Total odds document.getElementById('total-odds').textContent = betslip.totalOdds ? `x${Number(betslip.totalOdds).toFixed(2)}` : '–'; // Match bar if (resolution) { const bar = document.getElementById('match-bar'); bar.style.display = ''; const pct = Math.round(matchRate * 100); const cls = pct === 100 ? 'full' : pct >= 50 ? 'partial' : 'low'; document.getElementById('match-rate-text').textContent = `${matchedCount}/${sel.length}`; document.getElementById('match-rate-text').className = `match-rate ${cls}`; const fill = document.getElementById('progress-fill'); fill.style.width = pct + '%'; fill.className = `progress-fill ${cls}`; } // Selections const list = document.getElementById('selections-list'); list.innerHTML = ''; sel.forEach(s => { const div = document.createElement('div'); div.className = 'sel-item'; const eventDisplay = s.homeTeam && s.awayTeam ? `${s.homeTeam} vs ${s.awayTeam}` : s.eventName || 'Unknown Event'; const time = s.startTime ? new Date(s.startTime).toLocaleString('en-GB', { day: '2-digit', month: 'short', hour: '2-digit', minute: '2-digit' }) : ''; // Status dot const status = s.status || 'matched'; const dotCls = status === 'matched' ? 'matched' : (status === 'event_not_found' || status === 'unmatched') ? 'unmatched' : 'partial'; // Match note const noteText = s.note || ''; const noteClass = status === 'matched' ? 'ok' : ''; div.innerHTML = `
${escHtml(eventDisplay)}
${escHtml(s.market)}${time ? ' · ' + time : ''}
${s.odds ? Number(s.odds).toFixed(2) : '–'}
${s.selection ? `${escHtml(s.selection)}` : ''} ${noteText ? `
${escHtml(noteText)}
` : ''} `; list.appendChild(div); }); // Target link currentTargetLink = deeplink || ''; const linkEl = document.getElementById('target-link'); const linkText = document.getElementById('target-link-text'); if (deeplink) { linkEl.href = deeplink; linkEl.style.display = ''; linkText.textContent = (targetPlatform === 'stake' ? 'Open Stake.com' : 'Open SportyBet') + ' →'; } else { linkEl.style.display = 'none'; } document.getElementById('action-label').textContent = targetPlatform === 'stake' ? 'Continue on Stake' : 'Continue on SportyBet'; // Note box const noteEl = document.getElementById('note-box'); if (resolution && matchedCount < sel.length) { const missed = sel.length - matchedCount; noteEl.innerHTML = `${matchedCount}/${sel.length} selections matched automatically. ` + `${missed} could not be found on the target platform — ` + `add those manually. Green dot = auto-added to betslip.`; } else if (resolution && matchedCount === sel.length) { noteEl.innerHTML = `All ${sel.length} selections matched! Click the link to open your pre-filled betslip.`; } else { noteEl.innerHTML = 'Use the selections list to recreate your bet on the target platform.'; } document.getElementById('result-card').classList.add('visible'); document.getElementById('result-card').scrollIntoView({ behavior: 'smooth', block: 'nearest' }); } function copyLink() { if (!currentTargetLink) return; navigator.clipboard.writeText(currentTargetLink).then(() => { const btn = document.getElementById('copy-btn'); btn.classList.add('copied'); btn.innerHTML = '✓ Copied!'; setTimeout(() => { btn.classList.remove('copied'); btn.innerHTML = '📋 Copy'; }, 2000); }); } function setLoading(on) { document.getElementById('loader').classList.toggle('visible', on); document.getElementById('convert-btn').disabled = on; document.getElementById('convert-btn').textContent = on ? 'Fetching…' : 'Convert'; } function hideResult() { document.getElementById('result-card').classList.remove('visible'); } function showError(msg) { const box = document.getElementById('error-box'); box.textContent = '⚠ ' + msg; box.classList.add('visible'); } function hideError() { document.getElementById('error-box').classList.remove('visible'); } function escHtml(str) { return String(str) .replace(/&/g,'&').replace(//g,'>').replace(/"/g,'"'); }
